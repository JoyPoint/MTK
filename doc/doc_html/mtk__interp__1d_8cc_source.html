137 mtk::DenseMatrix mtk::Interp1D::ReturnAsDenseMatrix(
138  const UniStgGrid1D &grid) const {
139 
140  int nn{grid.num_cells_x()}; // Number of cells on the grid.
141 
142  #if MTK_PERFORM_PREVENTIONS
143  mtk::Tools::Prevent(nn <= 0, __FILE__, __LINE__, __func__);
144  #endif
145 
146  int gg_num_rows{}; // Number of rows.
147  int gg_num_cols{}; // Number of columns.
148 
149  if (dir_interp_ == mtk::SCALAR_TO_VECTOR) {
150  gg_num_rows = nn + 1;
151  gg_num_cols = nn + 2;
152  } else {
153  gg_num_rows = nn + 2;
154  gg_num_cols = nn + 1;
155  }
156 
157  // Output matrix featuring sizes for gradient operators.
158 
159  mtk::DenseMatrix out(gg_num_rows, gg_num_cols);
160 
162 
163  out.SetValue(0, 0, mtk::kOne);
164 
166 
167  for (auto ii = 1; ii < gg_num_rows - 1; ++ii) {
168  for(auto jj = ii ; jj < order_accuracy_ + ii; ++jj) {
169  out.SetValue(ii, jj, mtk::kOne/order_accuracy_);
170  }
171  }
172 
174 
175  out.SetValue(gg_num_rows - 1, gg_num_cols - 1, mtk::kOne);
176 
177  return out;
178 }
