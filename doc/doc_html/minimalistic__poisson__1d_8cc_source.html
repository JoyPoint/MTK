97 mtk::Real Source(mtk::Real xx) {
98  mtk::Real lambda = -1.0;
99  return lambda*lambda*exp(lambda*xx)/(exp(lambda) - 1.0);
100 }

102 mtk::Real KnownSolution(mtk::Real xx) {
103  mtk::Real lambda = -1.0;
104  return (exp(lambda*xx) - 1.0)/(exp(lambda) - 1.0);
105 }
