166 bool mtk::RobinBCDescriptor1D::ImposeOnLaplacianMatrix(
167  const mtk::Lap1D &lap,
168  mtk::DenseMatrix &matrix,
169  const mtk::Real &time) const {
170 
171  #ifdef MTK_PERFORM_PREVENTIONS
172  mtk::Tools::Prevent(highest_order_diff_west_ == -1,
173  __FILE__, __LINE__, __func__);
174  mtk::Tools::Prevent(highest_order_diff_east_ == -1,
175  __FILE__, __LINE__, __func__);
176  mtk::Tools::Prevent(matrix.num_rows() == 0, __FILE__, __LINE__, __func__);
177  mtk::Tools::Prevent(matrix.num_cols() == 0, __FILE__, __LINE__, __func__);
178  #endif
179 
182  matrix.SetValue(0, 0, (west_coefficients_[0])(time));
183 
185  matrix.SetValue(matrix.num_rows() - 1,
186  matrix.num_cols() - 1,
187  (east_coefficients_[0])(time));
188 
190  if (highest_order_diff_west_ > 0) {
191 
193  mtk::Grad1D grad;
194  if (!grad.ConstructGrad1D(lap.order_accuracy(),
195  lap.mimetic_threshold())) {
196  return false;
197  }
198 
200 
204  mtk::DenseMatrix coeffs(grad.mim_bndy());
205 
206  mtk::Real idx = mtk::kOne/lap.delta();
207 
209  for (int ii = 0; ii < coeffs.num_cols(); ++ii) {
211  mtk::Real aux{idx*coeffs.GetValue(0, ii)};
214  mtk::Real unit_normal{-mtk::kOne};
215  aux *= unit_normal*(west_coefficients_[1])(time);
217  matrix.SetValue(0, ii, matrix.GetValue(0, ii) + aux);
218  }
219 
221 
226 
227  for (int ii = 0; ii < coeffs.num_cols(); ++ii) {
229  mtk::Real aux{idx*coeffs.GetValue(0, ii)};
233  mtk::Real unit_normal{mtk::kOne};
234  aux *= -unit_normal*(east_coefficients_[1])(time);
236  matrix.SetValue(matrix.num_rows() - 1,
237  matrix.num_rows() - 1 - ii,
238  matrix.GetValue(matrix.num_rows() - 1,
239  matrix.num_rows() - 1 -ii) + aux);
240  }
241  }
242 
243  return true;
244 }
