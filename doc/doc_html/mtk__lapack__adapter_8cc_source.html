583  dgeqrf_(&aaT_num_rows, &aaT_num_cols, aa.data(), &aaT_num_rows,

588  tau,
589  work, &lwork, &info);
590  #endif
591 
592  #if MTK_DEBUG_LEVEL > 0
593  if (info == 0) {
594  lwork = (int) work[0];
595  } else {
596  std::cerr << "Could not get value for lwork on line " << __LINE__ - 5 <<

604  << std::endl;
605  #endif
606 
607  delete [] work;
608  work = nullptr;
609 
610  // Once we know lwork, we can actually invoke the factorization:
611  try {
612  work = new mtk::Real [lwork];
613  } catch (std::bad_alloc &memory_allocation_exception) {
614  std::cerr << "Memory allocation exception on line " << __LINE__ - 3 <<
615  std::endl;

617  }
618  memset(work, mtk::kZero, sizeof(work[0])*lwork);
619

621 
622  try {

624  } catch (std::bad_alloc &memory_allocation_exception) {
625  std::cerr << "Memory allocation exception on line " << __LINE__ - 3 <<

634  #else
635  fgeqrf_(&aaT_num_rows, &aaT_num_cols, aa.data(), &aaT_num_rows,
636  tau, work, &lwork, &info);
637  #endif
638 
639  if (!info) {
640  #if MTK_DEBUG_LEVEL > 0
641  std::cout << "QR factorization completed!" << std::endl << std::endl;
642  #endif
643  } else {
644  std::cerr << "Error solving system! info = " << info << std::endl;
645  std::cerr << "Exiting..." << std::endl;
646  }
647 
648  #if MTK_DEBUG_LEVEL > 0

651  #endif
652

655  // the orthogonal matrix Q as a product of min(aa_num_rows,aa_num_cols)

657  // value for lwork that is used.
658

660 
661  bool transpose{false};
662

664 
665  #if MTK_DEBUG_LEVEL > 0

668  #endif

672

675

679  std::cerr << "Memory allocation exception on line " << __LINE__ - 3 <<
680  std::endl;
681  std::cerr << memory_allocation_exception.what() <<
682  std::endl;
683  }
684  memset(work, mtk::kZero, sizeof(work[0])*1);
685

687  char trans_{'N'};
688 
689  int aux = QQ_.num_rows();
690 
691  #ifdef MTK_PRECISION_DOUBLE
692  dormqr_(&side_, &trans_,
693  &aa_num_cols, &aa_num_cols, &ltau, aa.data(), &aaT_num_rows, tau,
694  QQ_.data(), &aux, work, &lwork, &info);
695  #else
696  formqr_(&side_, &trans_,
697  &aa_num_cols, &aa_num_cols, &ltau, aa.data(), &aaT_num_rows, tau,
698  QQ_.data(), &aux, work, &lwork, &info);
699  #endif
700

707  }
708  #endif
709

711  std::cout << "lwork = " << std::endl << std::setw(12) << lwork <<
712  std::endl << std::endl;
713  #endif
714 
715  delete[] work;
716  work = nullptr;
717 
718  try {
719  work = new mtk::Real[lwork];
720  } catch (std::bad_alloc &memory_allocation_exception) {
721  std::cerr << "Memory allocation exception on line " << __LINE__ - 3 <<
722  std::endl;
723  std::cerr << memory_allocation_exception.what() << std::endl;
724  }
725  memset(work, mtk::kZero, sizeof(work[0])*lwork);
726 
727  #ifdef MTK_PRECISION_DOUBLE
728  dormqr_(&side_, &trans_,
729  &aa_num_cols, &aa_num_cols, &ltau, aa.data(), &aaT_num_rows, tau,
730  QQ_.data(), &aux, work, &lwork, &info);
731  #else
732  formqr_(&side_, &trans_,
733  &aa_num_cols, &aa_num_cols, &ltau, aa.data(), &aaT_num_rows, tau,
734  QQ_.data(), &aux, work, &lwork, &info);
735  #endif
736 
737  if (!info) {
738  #if MTK_DEBUG_LEVEL>0
739  std::cout << "Q matrix successfully assembled!" << std::endl << std::endl;
740  #endif
741  } else {
742  std::cerr << "Something went wrong solving system! info = " << info <<
743  std::endl;
744  std::cerr << "Exiting..." << std::endl;
745  }
746 
747  delete[] work;
748  work = nullptr;
749 
750  delete[] tau;
751  tau = nullptr;
752 
753  return QQ_;

756 int mtk::LAPACKAdapter::SolveRectangularDenseSystem(const mtk::DenseMatrix &aa,
757  mtk::Real *ob_,
758  int ob_ld_) {
759 
760  // We first invoke the solver to query for the value of lwork. For this,
761  // we must at least allocate enough space to allow access to WORK(1), or
762  // work[0]:
763 
764  // If LWORK = -1, then a workspace query is assumed; the routine only
765  // calculates the optimal size of the WORK array, returns this value as
766  // the first entry of the WORK array, and no error message related to
767  // LWORK is issued by XERBLA.
768 
769  mtk::Real *work{}; // Work array.
770 
771  try {
772  work = new mtk::Real[1];
773  } catch (std::bad_alloc &memory_allocation_exception) {
774  std::cerr << "Memory allocation exception on line " << __LINE__ - 3 << std::endl;
775  std::cerr << memory_allocation_exception.what() << std::endl;
776  }
777  memset(work, mtk::kZero, sizeof(work[0])*1);
778 
779  char trans_{'N'};
780  int nrhs_{1};
781  int info{0};
782  int lwork{-1};
783 
784  int AA_num_rows_ = aa.num_cols();
785  int AA_num_cols_ = aa.num_rows();
786  int AA_ld_ = std::max(1,aa.num_cols());
787 
788  #ifdef MTK_PRECISION_DOUBLE
789  dgels_(&trans_, &AA_num_rows_, &AA_num_cols_, &nrhs_, aa.data(), &AA_ld_,
790  ob_, &ob_ld_,
791  work, &lwork, &info);
792  #else
793  sgels_(&trans_, &AA_num_rows_, &AA_num_cols_, &nrhs_, aa.data(), &AA_ld_,
794  ob_, &ob_ld_,
795  work, &lwork, &info);
796  #endif
797 
798  if (info == 0) {
799  lwork = (int) work[0];
800  } else {
801  std::cerr << "Could not get value for lwork on line " << __LINE__ - 2 <<
802  std::endl;
803  std::cerr << "Exiting..." << std::endl;
804  return info;
805  }
806 
807  #if MTK_DEBUG_LEVEL > 0
808  std::cout << "lwork = " << std::endl << std::setw(12)<< lwork <<
809  std::endl << std::endl;
810  #endif
811 
812  // We then use lwork's new value to create the work array:
813  delete[] work;
814  work = nullptr;
815 
816  try {
817  work = new mtk::Real[lwork];
818  } catch (std::bad_alloc &memory_allocation_exception) {
819  std::cerr << "Memory allocation exception on line " << __LINE__ - 3 << std::endl;
820  std::cerr << memory_allocation_exception.what() << std::endl;
821  }
822  memset(work, 0.0, sizeof(work[0])*lwork);
823 
824  // We now invoke the solver again:
825  #ifdef MTK_PRECISION_DOUBLE
826  dgels_(&trans_, &AA_num_rows_, &AA_num_cols_, &nrhs_, aa.data(), &AA_ld_,
827  ob_, &ob_ld_,
828  work, &lwork, &info);
829  #else
830  sgels_(&trans_, &AA_num_rows_, &AA_num_cols_, &nrhs_, aa.data(), &AA_ld_,
831  ob_, &ob_ld_,
832  work, &lwork, &info);
833  #endif
834 
835  delete [] work;
836  work = nullptr;
837 
838  return info;
839 }
