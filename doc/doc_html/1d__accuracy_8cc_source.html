71 mtk::Real Polynomial(const mtk::Real &xx, const std::vector<mtk::Real> &pp) {
72 
73  mtk::Real sum{};
74 
75  mtk::Real kk{pp[0]};
76 
77  for (int ii = 0; ii <= kk; ++ii) {
78  sum += pow(xx, ii);
79  }
80  return sum;
81 }
