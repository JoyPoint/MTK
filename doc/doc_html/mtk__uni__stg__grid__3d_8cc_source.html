197  mtk::Tools::Prevent(top_bndy <= bottom_bndy,

201

318 void mtk::UniStgGrid3D::BindScalarField(
319  mtk::Real (*ScalarField)(const mtk::Real &xx,
320  const mtk::Real &yy,
321  const mtk::Real &zz)) {
322 
323  #ifdef MTK_PERFORM_PREVENTIONS
324  mtk::Tools::Prevent(nature_ != mtk::SCALAR, __FILE__, __LINE__, __func__);
325  #endif
326 
328 
329  discrete_domain_x_.reserve(num_cells_x_ + 2);
330 
331  discrete_domain_x_.push_back(west_bndy_);
332  #ifdef MTK_PRECISION_DOUBLE
333  auto first_center = west_bndy_ + delta_x_/2.0;
334  #else
335  auto first_center = west_bndy_ + delta_x_/2.0f;
336  #endif
337  discrete_domain_x_.push_back(first_center);
338  for (auto ii = 1; ii < num_cells_x_; ++ii) {
339  discrete_domain_x_.push_back(first_center + ii*delta_x_);
340  }
341  discrete_domain_x_.push_back(east_bndy_);
342 
344 
345  discrete_domain_y_.reserve(num_cells_y_ + 2);
346 
347  discrete_domain_y_.push_back(south_bndy_);
348  #ifdef MTK_PRECISION_DOUBLE
349  first_center = south_bndy_ + delta_x_/2.0;
350  #else
351  first_center = south_bndy_ + delta_x_/2.0f;
352  #endif
353  discrete_domain_y_.push_back(first_center);
354  for (auto ii = 1; ii < num_cells_y_; ++ii) {
355  discrete_domain_y_.push_back(first_center + ii*delta_y_);
356  }
357  discrete_domain_y_.push_back(north_bndy_);
358 
360 
361  discrete_domain_z_.reserve(num_cells_z_ + 2);
362 
363  discrete_domain_z_.push_back(bottom_bndy_);
364  first_center = bottom_bndy_ + delta_z_/mtk::kTwo;
365  discrete_domain_z_.push_back(first_center);
366  for (auto ii = 1; ii < num_cells_z_; ++ii) {
367  discrete_domain_z_.push_back(first_center + ii*delta_z_);
368  }
369  discrete_domain_z_.push_back(top_bndy_);
370 
372 
373  int aux{(num_cells_x_ + 2)*(num_cells_y_ + 2)*(num_cells_z_ + 2)};
374 
375  discrete_field_.reserve(aux);
376 
377  for (int kk = 0; kk < num_cells_z_ + 2; ++kk) {
378  for (int ii = 0; ii < num_cells_y_ + 2; ++ii) {
379  for (int jj = 0; jj < num_cells_x_ + 2; ++jj) {
380  #if MTK_VERBOSE_LEVEL > 6
381  std::cout << "At z = " << discrete_domain_z_[kk] << ": Pushing value"
382  " for x = " << discrete_domain_x_[jj] << " y = " <<
383  discrete_domain_y_[ii] << std::endl;
384  #endif
385  discrete_field_.push_back(ScalarField(discrete_domain_x_[jj],
386  discrete_domain_y_[ii],
387  discrete_domain_z_[kk]));
388  }
389  }
390  }
391 }

434  std::string space_name_x,
435  std::string space_name_y,
436  std::string space_name_z,

438 
439  std::ofstream output_dat_file; // Output file.

455  output_dat_file << discrete_domain_x_[jj] << ' ' <<
456  discrete_domain_y_[ii] << ' ' << discrete_domain_z_[kk] << ' ' <<
457  discrete_field_[idx] << std::endl;
458  idx++;
459  }
460  }
461  }
462 
463  } else {
464  output_dat_file << "# " << space_name_x << ' ' << space_name_y << ' ' <<
465  space_name_z << ' ' << field_name << std::endl;
466 
467  }
468 
469  output_dat_file.close();
