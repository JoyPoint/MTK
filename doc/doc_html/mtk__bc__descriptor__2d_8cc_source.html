181 void mtk::BCDescriptor2D::ImposeOnLaplacianMatrix(
182  const mtk::UniStgGrid2D &grid,
183  mtk::DenseMatrix &matrix) const {
184 
185  #if MTK_DEBUG_LEVEL > 0
186  mtk::Tools::Prevent(grid.nature() != mtk::SCALAR,
187  __FILE__, __LINE__, __func__);
188  mtk::Tools::Prevent(grid.num_cells_x() == 0, __FILE__, __LINE__, __func__);
189  mtk::Tools::Prevent(grid.num_cells_y() == 0, __FILE__, __LINE__, __func__);
190  mtk::Tools::Prevent(matrix.num_rows() == 0, __FILE__, __LINE__, __func__);
191  mtk::Tools::Prevent(matrix.num_cols() == 0, __FILE__, __LINE__, __func__);
192  #endif
193 
196 
197  bool generate_space{false};
198 
199  if (!grid.Bound()) {
200  generate_space = true;
201  }
202 
204 
205  if (generate_space) {
206  // For the south-west corner:
207  mtk::Real cc{};
208  for (unsigned int jj = 0; jj < south_coefficients_.size(); ++jj) {
209  // Evaluate the coefficient on the grid.
210  cc += (south_coefficients_[jj])(grid.west_bndy(), grid.south_bndy());
211  }
212  matrix.SetValue(0, 0, cc);
213 
214  // Compute first centers per dimension.
215  #ifdef MTK_PRECISION_DOUBLE
216  auto first_center_x = grid.west_bndy() + grid.delta_x()/2.0;
217  #else
218  auto first_center_x = grid.west_bndy() + grid.delta_x()/2.0f;
219  #endif
220 
221  #ifdef MTK_PRECISION_DOUBLE
222  auto first_center_y = grid.south_bndy() + grid.delta_y()/2.0;
223  #else
224  auto first_center_y = grid.south_bndy() + grid.delta_y()/2.0f;
225  #endif
226 
227  // For each entry on the diagonal (south boundary):
228  for (int ii = 0; ii < grid.num_cells_x(); ++ii) {
229  // Evaluate next set spatial coordinates to evaluate the coefficient at.
230  mtk::Real xx = first_center_x + ii*grid.delta_x();
231  mtk::Real yy = first_center_y + ii*grid.delta_y();
232 
233  cc = 0;
234  // For each coefficient to sum:
235  for (unsigned int jj = 0; jj < south_coefficients_.size(); ++jj) {
236  // Evaluate the coefficient on the grid and accumulate.
237  cc += (south_coefficients_[jj])(xx, yy);
238  }
239  matrix.SetValue(ii + 1, ii + 1, cc);
240  }
241 
242  // For the south-east corner:
243  cc = 0;
244  for (unsigned int jj = 0; jj < south_coefficients_.size(); ++jj) {
245  // Evaluate the coefficient on the grid.
246  cc += (south_coefficients_[jj])(grid.east_bndy(), grid.south_bndy());
247  }
248  matrix.SetValue(grid.num_cells_x() + 1, grid.num_cells_x() + 1, cc);
249 
250  } else {
251  // For each entry on the diagonal:
252  for (int ii = 0; ii < grid.num_cells_x() + 2; ++ii) {
253  // For each coefficient to sum:
254  mtk::Real xx{(grid.discrete_domain_x())[ii]};
255  mtk::Real yy{(grid.discrete_domain_y())[ii]};
256  mtk::Real cc{};
257  for (unsigned int jj = 0; jj < south_coefficients_.size(); ++jj) {
258  // Evaluate the coefficient on the grid.
259  cc += (south_coefficients_[jj])(xx,yy);
260  }
261  matrix.SetValue(ii, ii, cc);
262  }
263  }
264 
265 
267 }
