209 void mtk::BCDescriptor2D::ImposeOnSouthBoundary(
210  const mtk::UniStgGrid2D &grid,
211  mtk::DenseMatrix &matrix,
212  const int &order_accuracy) const {
213 
214  // At this point we have all of the information we need to fully impose the
215  // south boundary condition:
216  // 1. We have the collection of coefficients. The size of this collection
217  // tells us the type of BC for this boundary.
218  // 2. We have the grid that we can use to evaluate the coefficients at.
219  // 3. We have the matrix where to place them.
220 
221  // For now, we are sure that we will NOT have more than 2 coefficients per
222  // boundary. That is, we only support Robin type FOR NOW.
223 
224  if (generate_space_) {
225 
227 
228  // For the south-west corner:
229  auto cc = (south_coefficients_[0])(grid.west_bndy(), grid.south_bndy());
230 
231  #if MTK_DEBUG_LEVEL > 0
232  std::cout << "Matrix has " << matrix.num_rows() << " rows and " <<
233  matrix.num_cols() << " columns." << std::endl;
234  std::cout << "Setting at " << 0 << ' ' << 0 << std::endl;
235  #endif
236 
237  matrix.SetValue(0, 0, cc);
238 
239  // Compute first centers per dimension.
240  auto first_center_x = grid.west_bndy() + grid.delta_x()/mtk::kTwo;
241 
242  // For each entry on the diagonal (south boundary):
243  for (int ii = 0; ii < grid.num_cells_x(); ++ii) {
244  // Evaluate next set spatial coordinates to evaluate the coefficient.
245  mtk::Real xx = first_center_x + ii*grid.delta_x();
246  // Evaluate and assign the Dirichlet coefficient.
247  cc = (south_coefficients_[0])(xx, grid.south_bndy());
248 
249  #if MTK_DEBUG_LEVEL > 0
250  std::cout << "Setting at " << ii + 1 << ' ' << ii + 1 << std::endl;
251  #endif
252 
253  matrix.SetValue(ii + 1, ii + 1, cc);
254  }
255 
256  // For the south-east corner:
257  cc = (south_coefficients_[0])(grid.east_bndy(), grid.south_bndy());
258 
259  #if MTK_DEBUG_LEVEL > 0
260  std::cout << "Setting at " << grid.num_cells_x() + 1 << ' ' <<
261  grid.num_cells_x() + 1 << std::endl;
262  #endif
263 
264  matrix.SetValue(grid.num_cells_x() + 1, grid.num_cells_x() + 1, cc);
265 
267 
269  } else {
270 
272 
273  // For each entry on the diagonal:
274  for (int ii = 0; ii < grid.num_cells_x() + 2; ++ii) {
275  // Evaluate next set spatial coordinates to evaluate the coefficient.
276  mtk::Real xx{(grid.discrete_domain_x())[ii]};
277  // Evaluate and assign the Dirichlet coefficient.
278  mtk::Real cc = (south_coefficients_[0])(xx,grid.south_bndy());
279  matrix.SetValue(ii, ii, cc);
280  }
281 
283  }
