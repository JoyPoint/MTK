95 mtk::Real Source(const mtk::Real &xx, const mtk::Real &yy) {
96 
97  mtk::Real x_squared{xx*xx};
98  mtk::Real y_squared{yy*yy};
99  mtk::Real aux{-0.5*(x_squared + y_squared)};
100 
101  return xx*yy*exp(aux)*(x_squared + y_squared - 6.0);
102 }
