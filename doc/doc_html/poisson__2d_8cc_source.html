102 mtk::Real Source(const mtk::Real &xx, const mtk::Real &yy) {
103 
104  mtk::Real x_squared{xx*xx};
105  mtk::Real y_squared{yy*yy};
106  mtk::Real aux{-0.5*(x_squared + y_squared)};
107 
108  return xx*yy*exp(aux)*(x_squared + y_squared - 6.0);
109 }
