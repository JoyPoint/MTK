378 void mtk::BLASAdapter::RealDenseMV(mtk::Real &alpha,
379  mtk::DenseMatrix &aa,
380  mtk::Real *xx,
381  mtk::Real &beta,
382  mtk::Real *yy) {
383 
384  // Make sure input matrices are row-major ordered.
385 
386  if (aa.matrix_properties().ordering() == mtk::COL_MAJOR) {
387  aa.OrderRowMajor();
388  }
389 
390  char transa{'T'}; // State that now, the input WILL be in row-major ordering.
391 
392  int mm{aa.num_rows()}; // Rows of aa.
393  int nn{aa.num_cols()}; // Columns of aa.
394  int lda{(aa.matrix_properties()).ld()}; // Leading dimension.
395  int incx{1}; // Increment of values in x.
396  int incy{1}; // Increment of values in y.
397 
398  std::swap(mm,nn);
399  #ifdef MTK_PRECISION_DOUBLE
400  dgemv_(&transa, &mm, &nn, &alpha, aa.data(), &lda,
401  xx, &incx, &beta, yy, &incy);
402  #else
403  sgemv_(&transa, &mm, &nn, &alpha, aa.data(), &lda,
404  xx, &incx, &beta, yy, &incy);
405  #endif
406  std::swap(mm,nn);
407 }
