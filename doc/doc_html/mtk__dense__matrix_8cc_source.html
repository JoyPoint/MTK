463 mtk::DenseMatrix mtk::DenseMatrix::Kron(const mtk::DenseMatrix &aa,
464  const mtk::DenseMatrix &bb) {
465 
466  int row_offset{}; // Offset for rows.
467  int col_offset{}; // Offset for rows.
468 
469  mtk::Real aa_factor{}; // Used in computation.
470 
471  // Auxiliary variables:
472  auto aux1 = aa.matrix_properties_.num_rows()*bb.matrix_properties_.num_rows();
473  auto aux2 = aa.matrix_properties_.num_cols()*bb.matrix_properties_.num_cols();
474 
475  mtk::DenseMatrix output(aux1,aux2); // Output matrix.
476 
477  int kk_num_cols{output.matrix_properties_.num_cols()}; // Aux.
478 
479  auto mm = aa.matrix_properties_.num_rows(); // Rows of aa.
480  auto nn = aa.matrix_properties_.num_cols(); // Cols of aa.
481  auto pp = bb.matrix_properties_.num_rows(); // Rows of bb.
482  auto qq = bb.matrix_properties_.num_cols(); // Cols of bb.
483 
484  for (auto ii = 0; ii < mm; ++ii) {
485  row_offset = ii*pp;
486  for (auto jj = 0; jj < nn; ++jj) {
487  col_offset = jj*qq;
488  aa_factor = aa.data_[ii*nn + jj];
489  for (auto ll = 0; ll < pp; ++ll) {
490  for (auto oo = 0; oo < qq; ++oo) {
491  auto index = (ll + row_offset)*kk_num_cols + (oo + col_offset);
492  output.data_[index] = aa_factor*bb.data_[ll*qq + oo];
493  }
494  }
495  }
496  }
497 
498  output.matrix_properties_.set_storage(mtk::DENSE);
499  output.matrix_properties_.set_ordering(mtk::ROW_MAJOR);
500 
501  return output;
502 }
