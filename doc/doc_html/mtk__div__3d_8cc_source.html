78  int order_accuracy,

82  int num_cells_y = grid.num_cells_y();
83  int num_cells_z = grid.num_cells_z();

85  int mx = num_cells_x + 1; // Dx vertical dimension.

88  int ny = num_cells_y + 2; // Dy horizontal dimension.
89  int mz = num_cells_z + 1; // Dz vertical dimension.
90  int nz = num_cells_z + 2; // Dz horizontal dimension.
91 
92  mtk::Div1D div;
93 
94  bool info = div.ConstructDiv1D(order_accuracy, mimetic_threshold);
95 
96  #ifdef MTK_PERFORM_PREVENTIONS
97  if (!info) {
98  std::cerr << "Mimetic div could not be built." << std::endl;
99  return info;
100  }
101  #endif
102 
103  auto west = grid.west_bndy();
104  auto east = grid.east_bndy();
105  auto south = grid.south_bndy();
106  auto north = grid.east_bndy();
107  auto bottom = grid.bottom_bndy();
108  auto top = grid.top_bndy();
109 
110  mtk::UniStgGrid1D grid_x(west, east, num_cells_x);
111  mtk::UniStgGrid1D grid_y(south, north, num_cells_y);
112  mtk::UniStgGrid1D grid_z(bottom, top, num_cells_z);
113 
114  mtk::DenseMatrix Dx(div.ReturnAsDenseMatrix(grid_x));
115  mtk::DenseMatrix Dy(div.ReturnAsDenseMatrix(grid_y));
116  mtk::DenseMatrix Dz(div.ReturnAsDenseMatrix(grid_z));
117 
118  bool padded{true};
119  bool transpose{false};
120 
121  mtk::DenseMatrix ix(num_cells_x, padded, transpose);
122  mtk::DenseMatrix iy(num_cells_y, padded, transpose);
123  mtk::DenseMatrix iz(num_cells_z, padded, transpose);
124 
126 
127  mtk::DenseMatrix aux1(mtk::DenseMatrix::Kron(iz, iy));
128  mtk::DenseMatrix dx(mtk::DenseMatrix::Kron(aux1, Dx));
129 
131 
132  mtk::DenseMatrix aux2(mtk::DenseMatrix::Kron(iz, Dy));
133  mtk::DenseMatrix dy(mtk::DenseMatrix::Kron(aux2, ix));
134 
136 
137  mtk::DenseMatrix aux3(mtk::DenseMatrix::Kron(Dz, iy));
138  mtk::DenseMatrix dz(mtk::DenseMatrix::Kron(aux3, ix));
139 
140  #if MTK_VERBOSE_LEVEL > 2
141  std::cout << "Dx: " << mx << " by " << nx << std::endl;
142  std::cout << "Ix: " << num_cells_x << " by " << nx << std::endl;
143  std::cout << "Dy: " << my << " by " << ny << std::endl;
144  std::cout << "Iy: " << num_cells_y << " by " << ny << std::endl;
145  std::cout << "Dz: " << mz << " by " << nz << std::endl;
146  std::cout << "Iz: " << num_cells_z << " by " << nz << std::endl;
147  #endif
148 
150 
151  int total_rows{nx*ny*nz};
152  int total_cols{mx*num_cells_y*num_cells_z +
153  num_cells_x*my*num_cells_z +
154  num_cells_x*num_cells_y*mz};
155 
156  #if MTK_VERBOSE_LEVEL > 2
157  std::cout << "Div 3D: " << total_rows << " by " << total_cols << std::endl;
158  #endif
159 
160  mtk::DenseMatrix d3d(total_rows, total_cols);
161 
162  for (auto ii = 0; ii < total_rows; ++ii) {
163 
164  for (auto jj = 0; jj < mx*num_cells_y*num_cells_z; ++jj) {
165  d3d.SetValue(ii, jj, dx.GetValue(ii, jj));
166  }
167 
168  int offset = mx*num_cells_y*num_cells_z;
169 
170  for(auto kk = 0; kk < num_cells_x*my*num_cells_z; ++kk) {
171  d3d.SetValue(ii, kk + offset, dy.GetValue(ii, kk));
172  }
173 
174  offset += num_cells_x*my*num_cells_z;
175 
176  for(auto ll = 0; ll < num_cells_x*num_cells_y*mz; ++ll) {
177  d3d.SetValue(ii, ll + offset, dz.GetValue(ii, ll));
178  }
179  }
180 
181  divergence_ = d3d;
182 
183  return info;
184 }
