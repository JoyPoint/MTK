176 void mtk::UniStgGrid1D::BindScalarField(
177  mtk::Real (*ScalarField)(mtk::Real xx)) {
178 
179  #if MTK_DEBUG_LEVEL > 0
180  mtk::Tools::Prevent(nature_ == mtk::VECTOR, __FILE__, __LINE__, __func__);
181  #endif
182 
184 
185  discrete_domain_x_.reserve(num_cells_x_ + 2);
186 
187  discrete_domain_x_.push_back(west_bndy_x_);
188  #ifdef MTK_PRECISION_DOUBLE
189  auto first_center = west_bndy_x_ + delta_x_/2.0;
190  #else
191  auto first_center = west_bndy_x_ + delta_x_/2.0f;
192  #endif
193  discrete_domain_x_.push_back(first_center);
194  for (auto ii = 1; ii < num_cells_x_; ++ii) {
195  discrete_domain_x_.push_back(first_center + ii*delta_x_);
196  }
197  discrete_domain_x_.push_back(east_bndy_x_);
198 
200 
201  discrete_field_u_.reserve(num_cells_x_ + 2);
202 
203  discrete_field_u_.push_back(ScalarField(west_bndy_x_));
204 
205  discrete_field_u_.push_back(ScalarField(first_center));
206  for (auto ii = 1; ii < num_cells_x_; ++ii) {
207  discrete_field_u_.push_back(ScalarField(first_center + ii*delta_x_));
208  }
209  discrete_field_u_.push_back(ScalarField(east_bndy_x_));
210 }
