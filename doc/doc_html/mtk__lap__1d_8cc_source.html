119 bool mtk::Lap1D::ConstructLap1D(int order_accuracy,
120  mtk::Real mimetic_threshold) {
121 
122  #if MTK_DEBUG_LEVEL > 0
123  mtk::Tools::Prevent(order_accuracy < 2, __FILE__, __LINE__, __func__);
124  mtk::Tools::Prevent((order_accuracy%2) != 0, __FILE__, __LINE__, __func__);
125  mtk::Tools::Prevent(mimetic_threshold <= mtk::kZero,
126  __FILE__, __LINE__, __func__);
127 
128  if (order_accuracy >= mtk::kCriticalOrderAccuracyDiv) {
129  std::cout << "WARNING: Numerical accuracy is high." << std::endl;
130  }
131 
132  std::cout << "order_accuracy_ = " << order_accuracy << std::endl;
133  std::cout << "mimetic_threshold_ = " << mimetic_threshold << std::endl;
134  #endif
135 
136  order_accuracy_ = order_accuracy;
137  mimetic_threshold_ = mimetic_threshold;
138 
140 
141  mtk::Grad1D grad; // Mimetic gradient.
142 
143  bool info = grad.ConstructGrad1D(order_accuracy_, mimetic_threshold_);
144 
145  if (!info) {
146  std::cerr << "Mimetic grad could not be built." << std::endl;
147  return false;
148  }
149 
151 
152  mtk::Div1D div; // Mimetic divergence.
153 
154  info = div.ConstructDiv1D(order_accuracy_, mimetic_threshold_);
155 
156  if (!info) {
157  std::cerr << "Mimetic div could not be built." << std::endl;
158  return false;
159  }
160 
162 
163  // Since these are mimetic operator, we must multiply the matrices arising
164  // from both the divergence and the Laplacian, in order to get the
165  // approximating coefficients for the Laplacian operator.
166 
167  // However, we must choose a grid that implied a step size of 1, so to get
168  // the approximating coefficients, without being affected from the
169  // normalization with respect to the grid.
170 
171  // Also, the grid must be of the minimum size to support the requested order
172  // of accuracy. We must please the divergence.
173 
174  mtk::UniStgGrid1D aux(mtk::kZero,
175  (mtk::Real) 3*order_accuracy_ - 1,
176  3*order_accuracy_ - 1);
177 
178  #if MTK_DEBUG_LEVEL > 0
179  std::cout << "aux =" << std::endl;
180  std::cout << aux << std::endl;
181  std::cout <<"aux.delta_x() = " << aux.delta_x() << std::endl;
182  std::cout << std::endl;
183  #endif
184 
185  mtk::DenseMatrix grad_m(grad.ReturnAsDenseMatrix(aux));
186 
187  #if MTK_DEBUG_LEVEL > 0
188  std::cout << "grad_m =" << std::endl;
189  std::cout << grad_m << std::endl;
190  #endif
191 
192  mtk::DenseMatrix div_m(div.ReturnAsDenseMatrix(aux));
193 
194  #if MTK_DEBUG_LEVEL > 0
195  std::cout << "div_m =" << std::endl;
196  std::cout << div_m << std::endl;
197  #endif
198 
202 
203  mtk::DenseMatrix lap; // Laplacian matrix to hold to computed coefficients.
204 
205  lap = mtk::BLASAdapter::RealDenseMM(div_m, grad_m);
206 
207  #if MTK_DEBUG_LEVEL > 0
208  std::cout << "lap =" << std::endl;
209  std::cout << lap << std::endl;
210  #endif
211 
213 
215 
216  // The output array will have this form:
217  // 1. The first entry of the array will contain the used order kk.
218  // 2. The second entry of the array will contain the collection of
219  // approximating coefficients for the interior of the grid.
220  // 3. The next entries will contain the collections of approximating
221  // coefficients for the west boundary of the grid.
222 
223  laplacian_length_ = 1 + (2*order_accuracy_ - 1) +
224  (order_accuracy_ - 1)*(2*order_accuracy_);
225 
226  #if MTK_DEBUG_LEVEL > 0
227  std::cout << "laplacian_length_ = " << laplacian_length_ << std::endl;
228  std::cout << std::endl;
229  #endif
230 
231  try {
232  laplacian_ = new mtk::Real[laplacian_length_];
233  } catch (std::bad_alloc &memory_allocation_exception) {
234  std::cerr << "Memory allocation exception on line " << __LINE__ - 3 <<
235  std::endl;
236  std::cerr << memory_allocation_exception.what() << std::endl;
237  }
238  memset(laplacian_, mtk::kZero, sizeof(laplacian_[0])*laplacian_length_);
239 
241 
242  laplacian_[0] = order_accuracy_;
243 
246 
247  for (auto ii = 0; ii < 2*order_accuracy_ - 1; ++ii) {
248  laplacian_[ii + 1] = lap.GetValue(1 + (order_accuracy_ - 1), ii + 1);
249  }
250 
252 
253  auto offset = 1 + (2*order_accuracy_ - 1);
254 
255  for (auto ii = 0; ii < order_accuracy_ - 1; ++ii) {
256  for (auto jj = 0; jj < 2*order_accuracy_; ++jj) {
257  laplacian_[offset + ii*(2*order_accuracy_) + jj] =
258  lap.GetValue(1 + ii,jj);
259  }
260  }
261 
262  return true;
263 }

265 mtk::DenseMatrix mtk::Lap1D::ReturnAsDenseMatrix(
266  const UniStgGrid1D &grid) const {
267 
268  int nn{grid.num_cells_x()}; // Number of cells on the grid.
269 
270  #if MTK_DEBUG_LEVEL > 0
271  mtk::Tools::Prevent(nn <= 0, __FILE__, __LINE__, __func__);
272  mtk::Tools::Prevent(nn < 3*order_accuracy_ - 1, __FILE__, __LINE__, __func__);
273  #endif
274 
275  mtk::DenseMatrix lap(nn + 2, nn + 2); // Laplacian matrix to be returned.
276 
277  mtk::Real idx{mtk::kOne/(grid.delta_x()*grid.delta_x())}; // Inverse of dx^2.
278 
280 
281  auto offset = (1 + 2*order_accuracy_ - 1);
282 
283  for (auto ii = 0; ii < order_accuracy_ - 1; ++ii) {
284  for (auto jj = 0; jj < 2*order_accuracy_; ++jj) {
285  lap.SetValue(1 + ii,
286  jj,
287  idx*laplacian_[offset + ii*2*order_accuracy_ + jj]);
288  }
289  }
290 
292 
293  offset = 1 + (order_accuracy_ - 1);
294 
295  int kk{1};
296  for (auto ii = order_accuracy_; ii <= nn - (order_accuracy_ - 1); ++ii) {
297  int mm{1};
298  for (auto jj = 0; jj < 2*order_accuracy_ - 1; ++jj) {
299  lap.SetValue(ii, jj + kk, idx*laplacian_[mm]);
300  mm = mm + 1;
301  }
302  kk = kk + 1;
303  }
304 
306 
307  offset = (1 + 2*order_accuracy_ - 1);
308 
309  auto aux = order_accuracy_ + (nn - 2*(order_accuracy_ - 1));
310 
311  auto ll = 1;
312  auto rr = 1;
313  for (auto ii = nn; ii > aux - 1; --ii) {
314  auto cc = 0;
315  for (auto jj = nn + 2 - 1; jj >= (nn + 2) - 2*order_accuracy_; --jj) {
316  lap.SetValue(ii, jj, lap.GetValue(rr,cc));
317  ++ll;
318  ++cc;
319  }
320  rr++;
321  }
322 
329 
330  return lap;
331 }
